Queue (Linked List Implementaion)

Algorithm

public class Queue {
								private Node front;
								private Node rear;
								private int size;
							
								public Queue() {
									this.front = null;
									this.rear = null;
									this.size = 0;
								}
							
								public void enqueue(int data) {
									Node newNode = new Node(data);
									if (rear == null) {
										front = rear = newNode;
									} else {
										rear.next = newNode;
										rear = newNode;
									}
									size++;
								}
							
								public int dequeue() {
									if (front == null) {
										System.out.println("Queue Underflow");
										return -1;
									}
									int data = front.data;
									front = front.next;
									if (front == null) {
										rear = null;
									}
									size--;
									return data;
								}
							
								public void clear() {
									front = rear = null;
									size = 0;
								}
							
								private class Node {
									int data;
									Node next;
							
									Node(int data) {
										this.data = data;
									}
								}
							}
Time Complexity: O(1)
Space Complexity: O(1)
